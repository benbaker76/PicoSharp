function all(t)
  if t == nil then
    return function() end
  end
  local ttype = type(t)
  if ttype == 'string' then
    local i = 0
    local tlen = #t
    return function()
      i = i + 1
      if i <= tlen then
        return sub(t, i, i)
      end
    end
  else
    local i = 0
    return function()
      i = i + 1
      return t[i]
    end
  end
end

function add(t, v)
  if type(t) == 'table' then
    t[#t+1] = v
    return v
  end
end

function foreach(c, f)
  if c ~= nil then
    for value in all(c) do
      f(value)
    end
  end
end


function mapdraw(...)
  map(table.unpack(arg))
end

function count(t, v)
  if type(t) ~= 'table' then
    return nil
  end
  if v == nil then
    return #t
  end
  local c = 0
  for i = 1, #t do
    if t[i] == v then
      c = c + 1
    end
  end
  return c
end

function del(t, v)
  if type(t) == 'table' then
    local found = false
    for i = 1, #t do
      if t[i] == v then
        found = true
      end
      if found then
        t[i] = t[i+1]
      end
    end
    if found then
      return v
    end
  end
end

function deli(t, i)
  if type(t) == 'table' then
    if i == nil then
       i = #t
    end
    if i < 1 or i > #t then
      return nil
    end
    local v = t[i]
    for j = i, #t do
      t[j] = t[j+1]
    end
    return v
  end
end

function cocreate(f)
  return coroutine.create(f)
end

function yield()
  coroutine.yield()
end
